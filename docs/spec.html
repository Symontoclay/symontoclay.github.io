Fact

back to top

COMPLETE ME!
FactDeclSet = FactDecl [{ FactDecl }] .
FactDecl = "{:" ( FactSection | FactSentence ) ":}" .
FactSection = ">:" "{" FactSentence "}" .
FactSentence = FactPredicate .
FactPredicate = ConceptIdentifier "(" FactPredicateParam [ { "," FactPredicateParam } ] ")" .
FactPredicateParam = ConceptIdentifier | InstanceIdentifier .
COMPLETE ME!

Logic rule

back to top

COMPLETE ME!
LogicRuleDeclSet = LogicRuleDecl [{ LogicRuleDecl }] .
LogicRuleDecl = "{:" LogicRulePrimarySection ImplicationSymbol LogicRuleSecondarySection ":}" .
LogicRulePrimarySection = ">:" "{" LogicRuleSentence "}" .
LogicRuleSecondarySection = "{" LogicRuleSentence "}" .
ImplicationSymbol = "->" .
LogicRuleSentence = LogicRulePredicate [ { "&" LogicRulePredicate } ] .
LogicRulePredicate = ConceptIdentifier "(" LogicRulePredicateParam [ { "," LogicRulePredicateParam } ] ")" .
LogicRulePredicateParam = ConceptIdentifier | InstanceIdentifier | LogicSentenceVarIdentifier .
