npc PeaceKeeper is [0.5] exampleClass
{
    on Init =>
    {
        "Begin" >> @>log;
        exampleClass is human >> @>log;
        exampleClass is not human >> @>log;
        use exampleClass is [0.5] human;
        exampleClass is human >> @>log;
        exampleClass is not human >> @>log;
        use exampleClass is not human;
        exampleClass is human >> @>log;
        exampleClass is not human >> @>log;
        use @@self is linux;
        @@self is linux >> @>log;
        "End" >> @>log;

    {: male(#Tom) :}

    {: {son($x, $y)} -> { male($x) & parent($y, $x)} :}
    {: { love($x, $y) } -> { help($x, $y) } :}

{
    on Init =>
    {
        "Begin" >> @>log;
